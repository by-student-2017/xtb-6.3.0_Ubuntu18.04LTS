# xtb-6.3.0_Ubuntu18.04LTS


-----
# Installation


1. sudo apt update


2. sudo apt install -y git


3. git clone https://github.com/by-student-2017/xtb-6.3.0_Ubuntu18.04LTS.git


4. mv xtb-6.3.0_Ubuntu18.04LTS xtb-6.3.0


5. cd xtb-6.3.0


6. chmod +x install.sh


7. ./install.sh
